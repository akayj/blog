*** Golang

#+BEGIN_SRC go
// waitSignal regist quit signals, a done channel is returned
// as signal receiver
func waitSignal() <-chan os.Signal {
    sigs := make(chan os.Signal, 1)
    done := make(chan os.Signal)

    signal.Notify(sigs, syscall.SIGINT, syscal.SIGTERM)

    go func() {
        sig := <- sigs
        done <- sig
    }()

    return done
}
#+END_SRC
